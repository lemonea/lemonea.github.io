<!DOCTYPE html>

<html>

<head>

    <title>Crypto Prices Loading...</title>

</head>

<body>

    <h2>Real-time Crypto Prices</h2>

    <div id="display">Loading...</div>

    

    <script>

        async function updateAll() {

            try {

                const ethResp = await fetch('https://api.binance.com/api/v3/ticker/price?symbol=ETHUSDT');

                const ethData = await ethResp.json();

                const btcResp = await fetch('https://api.binance.com/api/v3/ticker/price?symbol=BTCUSDT');

                const btcData = await btcResp.json();

                

                const ethPrice = parseFloat(ethData.price).toFixed(2);

                const btcPrice = parseFloat(btcData.price).toFixed(2);

                const time = new Date().toLocaleTimeString();

                

                // 方法1: 更新标题

                document.title = `ETH: $${ethPrice} | BTC: $${btcPrice}`;

                

                // 方法2: 重写整个body

                document.body.innerHTML = `

                    <h2>Crypto Prices (Updated: ${time})</h2>

                    <div>ETH: $${ethPrice}</div>

                    <div>BTC: $${btcPrice}</div>

                    <div>Last Update: ${time}</div>

                    <!-- CLAUDE_DATA: ETH:$${ethPrice}, BTC:$${btcPrice} -->

                `;

                

            } catch (error) {

                document.body.innerHTML = `<div>Error: ${error.message}</div>`;

            }

        }

        

        updateAll();

        setInterval(updateAll, 15000);

    </script>

</body>

</html>
